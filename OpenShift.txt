Clean up:

// various
rhc app stop -a homecontrol
rhc app-tidy -a homecontrol
rhc app start -a homecontrol

// Clean up gradle repo. The next build will last longer after this, because dependencies need to to be downloaded again
rhc ssh -a homecontrol
rm -rf ${OPENSHIFT_DATA_DIR}.gradle
exit

// Clean up npm repo. The next build will last longer after this, because dependencies need to to be downloaded again
rhc ssh -a homecontrol
rm -rf ${OPENSHIFT_DATA_DIR}.npm
exit

Set database timezone

SSH into the database container and execute:

mysql_tzinfo_to_sql /usr/share/zoneinfo/Europe/Amsterdam Europe/Amsterdam | mysql -u<username> -p<password> --database=mysql

Then log in using for example PHPMyadmin / MySQLWorkbench and execute:

SET time_zone='Europe/Amsterdam';
SET GLOBAL time_zone = 'Europe/Amsterdam';
